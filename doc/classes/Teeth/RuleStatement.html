    def function_body
      code = ""
      code += "  BEGIN(#{@begin});\n" if @begin
      if skip_line
        code += "  return EOF_KVPAIR;\n"
      else
        code += "  KVPAIR #{name.to_s} = {\"#{name.to_s}\", #{yytext_statement}};\n" +
        "  return #{name.to_s};\n"
      end
      code
    end
